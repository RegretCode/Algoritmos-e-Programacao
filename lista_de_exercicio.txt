#include <stdio.h>
#include <stdlib.h>

int main()
{
int a, b;

printf ("Digite um valor: \n");
scanf ("%d", &a);
printf ("Digite outro valor: \n");
scanf ("%d", &b);

if (a == b) 
{
printf ("Não são considerados valores iguais");
}
if (a < b)
{
printf ("Em ordem crescente %d e %d", a, b);

}
if (a > b)
{
printf ("Em ordem crescente %d e %d", b, a);

}
}
